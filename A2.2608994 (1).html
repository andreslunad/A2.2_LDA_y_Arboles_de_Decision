from sklearn.tree import DecisionTreeClassifier, plot_tree
from sklearn.model_selection import cross_val_score, LeaveOneOut
from sklearn.metrics import accuracy_score, f1_score
X_tree_train = X_trainvar  # Glucose y Resistin
X_tree_test = X_testvar    # Glucose y Resistin
y_tree_train = y_train

tree_initial = DecisionTreeClassifier(random_state=42)
tree_initial.fit(X_tree_train, y_tree_train)

# 2. Visualizar rbol inicial
plt.figure(figsize=(15, 10))
plot_tree(tree_initial, filled=True, feature_names=X_tree_train.columns.tolist(), class_names=['Saludable', 'Condicin'], rounded=True)
plt.title("rbol de Decisin - Sin Poda")
plt.show()

LOO= LeaveOneOut()
ccp_alphas= np.linspace(0.001, 0.1, 100)
cv_scores= []

for alpha in ccp_alphas:
    tree_temp = DecisionTreeClassifier(ccp_alpha=alpha, random_state=42)
    scores = cross_val_score(tree_temp, X_tree_train, y_tree_train, cv=loo, scoring='accuracy')
    cv_scores.append(np.mean(scores))

best_alpha = ccp_alphas[np.argmax(cv_scores)]
print(f"Mejor alpha encontrado: {best_alpha:.4f}")

# 4. Entrenar rbol podado con el mejor alpha
pruned_tree = DecisionTreeClassifier(ccp_alpha=best_alpha, random_state=42)

# 5. Visualizar rbol podado
plt.figure(figsize=(12, 8))
plot_tree(pruned_tree, filled=True, feature_names=X_tree_train.columns.tolist(), class_names=['Saludable', 'Condicin'], rounded=True, fontsize=10)

from sklearn.metrics import confusion_matrix, accuracy_score, f1_score, recall_score, precision_score
import pandas as pd

def classification_metrics(y_true, y_pred):
    tn, fp, fn, tp = confusion_matrix(y_true, y_pred).ravel()
    accuracy = accuracy_score(y_true, y_pred)
    sensitivity = recall_score(y_true, y_pred)         
    specificity = tn / (tn + fp) if (tn+fp)>0 else 0    
    precision = precision_score(y_true, y_pred)
    f1 = f1_score(y_true, y_pred)
    return {
        "Accuracy": accuracy,
        "Sensitivity": sensitivity,
        "Specificity": specificity,
        "Precision": precision,
        "F1-score": f1
    }
